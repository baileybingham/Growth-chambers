## Climate variables
## Data are aggregated to daily mean, minimum and maximum 
## Variable type: sensor_name
TMS_T1 = temperature degrees Celsius - soil 
TMS_T2 = temperature degrees Celsius - near soil
TMS_T3 = temperature degrees Celsius - aboveground
TMS_moist = gravimetric water content of soil
